```python
import os
import uuid
import asyncio
from typing import Dict, Any
from openai import AsyncOpenAI
from mcp.server.fastmcp import FastMCP

# =============================================================================
# INITIALIZATION & CONFIGURATION
# =============================================================================

# Initialize FastMCP server
mcp = FastMCP("Dynamic Supervisor")

# API Configuration: Use environment variable for Tailscale/Mesh VPN support
# Fallback to localhost:1234 if not provided
BASE_URL = os.getenv("LM_STUDIO_BASE_URL", "http://localhost:1234/v1")
client = AsyncOpenAI(base_url=BASE_URL, api_key="lm-studio")

# =============================================================================
# STATE MANAGEMENT (In-Memory)
# =============================================================================

# agents stores: { agent_id: {"role": str, "system_prompt": str} }
agents: Dict[str, Dict[str, str]] = {}

# tasks stores: { task_id: {"status": str, "result": str} }
# status can be: 'running', 'completed', or 'failed'
tasks: Dict[str, Dict[str, Any]] = {}

# =============================================================================
# SUB-AGENT LOGIC (Background Worker)
# =============================================================================

async def _run_agent_task(agent_id: str, task_id: str, prompt: str):
    """
    The background worker that performs the actual LLM inference.
    This runs concurrently without blocking the MCP server.
    """
    agent_profile = agents.get(agent_id)
    if not agent_profile:
        tasks[task_id] = {"status": "failed", "result": "Agent profile not found."}
        return

    try:
        # Format the chat messages using the agent's specific system prompt
        messages = [
            {"role": "system", "content": agent_profile["system_prompt"]},
            {"role": "user", "content": prompt}
        ]

        # Perform the asynchronous API call to LM Studio
        response = await client.chat.completions.create(
            model="local-model",  # LM Studio typically ignores the model name and uses the loaded one
            messages=messages,
            temperature=0.7,
        )

        # Extract result and update state
        result_text = response.choices[0].message.content
        tasks[task_id] = {
            "status": "completed",
            "result": result_text
        }

    except Exception as e:
        # Log the error into the task status for the supervisor to see
        tasks[task_id] = {
            "status": "failed",
            "result": f"Error during execution: {str(e)}"
        }

# =============================================================================
# MCP TOOL DEFINITIONS
# =============================================================================

@mcp.tool()
async def create_agent(role: str, system_prompt: str) -> str:
    """
    Generates a unique sub-agent with a specific role and system prompt.
    Returns the unique agent_id.
    """
    agent_id = f"agent_{uuid.uuid4().hex[:8]}"
    agents[agent_id] = {
        "role": role,
        "system_prompt": system_prompt
    }
    return agent_id

@mcp.tool()
async def assign_task(agent_id: str, prompt: str) -> str:
    """
    Assigns a task to an existing agent. This is non-blocking.
    Returns a task_id that can be used to track progress.
    """
    # Verify agent exists
    if agent_id not in agents:
        return f"Error: Agent {agent_id} does not exist."

    # Generate unique task ID
    task_id = f"task_{uuid.uuid4().hex[:8]}"
    
    # Initialize task status
    tasks[task_id] = {"status": "running", "result": ""}

    # Kick off the background task using asyncio.create_task
    # This allows the tool to return the task_id immediately
    asyncio.create_task(_run_agent_task(agent_id, task_id, prompt))

    return task_id

@mcp.tool()
async def check_task_status(task_id: str) -> str:
    """
    Returns the current status of a task ('running', 'completed', or 'failed').
    """
    task = tasks.get(task_id)
    if not task:
        return "Error: Task ID not found."
    
    return task["status"]

@mcp.tool()
async def retrieve_result(task_id: str) -> str:
    """
    Returns the generated text from the sub-agent if the task is completed.
    """
    task = tasks.get(task_id)
    if not task:
        return "Error: Task ID not found."
    
    if task["status"] == "running":
        return "Task is still running. Please check status later."
    
    if task["status"] == "failed":
        return f"Task failed. Error: {task.get('result')}"

    return task.get("result", "No result found.")

# =============================================================================
# ENTRY POINT
# =============================================================================

if __name__ == "__main__":
    # FastMCP handles the stdio transport and lifecycle automatically
    mcp.run()
```

Because this server is built using asynchronous Python (`asyncio`) and in-memory dictionaries, the answer depends on whether you are talking about **defined agents** or **active tasks**.

### 1. How many agents does the server support?
**Technically: Virtually unlimited (limited by RAM).**

*   **Storage:** The agents are stored in a standard Python dictionary (`agents = {}`). A dictionary can hold millions of entries without a performance hit.
*   **Memory Footprint:** Each agent only stores a small string for its `role` and `system_prompt`. Even if you created 100,000 agents, it would likely only consume a few dozen megabytes of RAM.
*   **The Real Limit:** You will run out of system memory (RAM) long before you run out of "slots" for agents.

---

### 2. What happens if the model is limited to $X$ concurrent predictions?
This is where the architecture hits a bottleneck. There is a distinction between **Defined Agents** and **Active Tasks**. You can have 1,000 agents, but if you try to run 1,000 tasks at once, you will hit the model's limit.

If your LM Studio instance (or the hardware running it) can only handle $X$ concurrent requests, one of two things will happen depending on how LM Studio is configured:

#### **Scenario A: The "Queue" Behavior (Most Common)**
If LM Studio (or the underlying web server like Uvicorn) has a request queue:
*   **What happens:** The MCP server will successfully "assign" the task and return a `task_id`.
*   **Status:** When you call `check_task_status`, it will return **`running`** for a very long time.
*   **Result:** The task isn't failing; it is simply sitting in a line waiting for its turn. Once the $X$ current tasks finish, the next one in line starts.

#### **Scenario B: The "Rejection" Behavior (Hard Limit)**
If LM Studio is configured to reject connections once the limit is reached (returning an HTTP `429 Too Many Requests` or `503 Service Unavailable` error):
*   **What happens:** The `await client.chat.completions.create(...)` call inside the sub-agent will immediately throw an exception.
*   **Status:** The `try...except` block in my code will catch this error.
*   **Result:** The task will move immediately to **`failed`**, and `retrieve_result` will tell you exactly what happened (e.g., *"Error during execution: 429 Too Many Requests"*).

---

### **Summary Table**

| Metric | Limit Type | Bottleneck |
| :--- | :--- | :--- |
| **Number of Agents** | **Storage** | System RAM (extremely high limit) |
| **Number of Tasks** | **Concurrency** | LM Studio's GPU/CPU & API capacity |

### **Pro-Engineer Tip: How to make it "Bulletproof"**
If you wanted to scale this to a massive production environment, you would implement a **Semaphore** in the code. 

Instead of just running `asyncio.create_task`, you would use:
`async with semaphore: await client.chat.completions.create(...)`

A **Semaphore** acts like a "gatekeeper" inside the MCP server. If you set the semaphore to `5`, the MCP server will only allow 5 tasks to actually attempt to hit LM Studio at any given time. Any additional tasks would wait inside the MCP server itself, preventing them from ever receiving an "Error" from the API and ensuring they all eventually finish successfully.<turn|>

To implement the Semaphore, I have modified the `_run_agent_task` function. I have also added a new environment variable, `MCP_CONCURRENCY_LIMIT`, so you can tune how many agents are allowed to talk to LM Studio at the exact same moment without needing to edit the code again.

### **What this change does:**
If you set the limit to `3` and you assign `10` tasks:
1.  The first `3` tasks will immediately start hitting the LM Studio API.
2.  The remaining `7` tasks will stay in the `running` state, but they will be "waiting in line" inside the MCP server.
3.  As soon as one task finishes, the next one in line automatically starts.
4.  **This prevents the 429 (Too Many Requests) or 503 errors you saw earlier.**

### **The Updated Code**

```python
import os
import uuid
import asyncio
from typing import Dict, Any
from openai import AsyncOpenAI
from mcp.server.fastmcp import FastMCP

# =============================================================================
# INITIALIZATION & CONFIGURATION
# =============================================================================

# Initialize FastMCP server
mcp = FastMCP("Dynamic Supervisor")

# API Configuration
BASE_URL = os.getenv("LM_STUDIO_BASE_URL", "http://localhost:1234/v1")
API_KEY = os.getenv("LM_STUDIO_API_KEY", "lm-studio")
client = AsyncOpenAI(base_url=BASE_URL, api_key=API_KEY)

# CONCURRENCY CONTROL
# This limits how many LLM API calls can happen at the exact same time.
# Default is 3. You can change this via environment variable.
CONCURRENCY_LIMIT = int(os.getenv("MCP_CONCURRENCY_LIMIT", "3"))
semaphore = asyncio.Semaphore(CONCURRENCY_LIMIT)

# =============================================================================
# STATE MANAGEMENT (In-Memory)
# =============================================================================

# agents stores: { agent_id: {"role": str, "system_prompt": str} }
agents: Dict[str, Dict[str, str]] = {}

# tasks stores: { task_id: {"status": str, "result": str} }
tasks: Dict[str, Dict[str, Any]] = {}

# =============================================================================
# SUB-AGENT LOGIC (Background Worker)
# =============================================================================

async def _run_agent_task(agent_id: str, task_id: str, prompt: str):
    """
    The background worker that performs the actual LLM inference.
    Uses a Semaphore to ensure we don't overwhelm the local LLM server.
    """
    agent_profile = agents.get(agent_id)
    if not agent_profile:
        tasks[task_id] = {"status": "failed", "result": "Agent profile not found."}
        return

    try:
        # The Semaphore acts as a gatekeeper. 
        # If the limit is reached, this 'async with' block will pause 
        # execution here until a 'slot' opens up.
        async with semaphore:
            # Format the chat messages using the agent's specific system prompt
            messages = [
                {"role": "system", "content": agent_profile["system_prompt"]},
                {"role": "user", "content": prompt}
            ]

            # Perform the asynchronous API call to LM Studio
            response = await client.chat.completions.create(
                model="local-model", 
                messages=messages,
                temperature=0.7,
            )

            # Extract result
            result_text = response.choices[0].message.content
            tasks[task_id] = {
                "status": "completed",
                "result": result_text
            }

    except Exception as e:
        # If the API call fails or a timeout occurs, log it to the task status
        tasks[task_id] = {
            "status": "failed",
            "result": f"Error during execution: {str(e)}"
        }

# =============================================================================
# MCP TOOL DEFINITIONS
# =============================================================================

@mcp.tool()
async def create_agent(role: str, system_prompt: str) -> str:
    """
    Generates a unique sub-agent with a specific role and system prompt.
    Returns the unique agent_id.
    """
    agent_id = f"agent_{uuid.uuid4().hex[:8]}"
    agents[agent_id] = {
        "role": role,
        "system_prompt": system_prompt
    }
    return agent_id

@mcp.tool()
async def assign_task(agent_id: str, prompt: str) -> str:
    """
    Assigns a task to an existing agent. This is non-blocking.
    Returns a task_id that can be used to track progress.
    """
    if agent_id not in agents:
        return f"Error: Agent {agent_id} does not exist."

    task_id = f"task_{uuid.uuid4().hex[:8]}"
    
    # Initialize task status as 'running' immediately.
    # If it hits the semaphore, it will stay 'running' while waiting in line.
    tasks[task_id] = {"status": "running", "result": ""}

    # Kick off the background task
    asyncio.create_task(_run_agent_task(agent_id, task_id, prompt))

    return task_id

@mcp.tool()
async def check_task_status(task_id: str) -> str:
    """
    Returns the current status of a task ('running', 'completed', or 'failed').
    """
    task = tasks.get(task_id)
    if not task:
        return "Error: Task ID not found."
    
    return task["status"]

@mcp.tool()
async def retrieve_result(task_id: str) -> str:
    """
    Returns the generated text from the sub-agent if the task is completed.
    """
    task = tasks.get(task_id)
    if not task:
        return "Error: Task ID not found."
    
    if task["status"] == "running":
        return "Task is still running (or waiting in the concurrency queue)."
    
    if task["status"] == "failed":
        return f"Task failed. Error: {task.get('result')}"

    return task.get("result", "No result found.")

# =============================================================================
# ENTRY POINT
# =============================================================================

if __name__ == "__main__":
    mcp.run()
```
