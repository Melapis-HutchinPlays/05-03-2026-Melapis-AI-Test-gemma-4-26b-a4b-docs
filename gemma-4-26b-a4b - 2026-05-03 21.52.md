```python
import os
import uuid
import asyncio
from typing import Dict, Any
from openai import AsyncOpenAI
from mcp.server.fastmcp import FastMCP

# =============================================================================
# INITIALIZATION & CONFIGURATION
# =============================================================================

# Initialize FastMCP server
mcp = FastMCP("Dynamic Supervisor")

# API Configuration: Use environment variable for Tailscale/Mesh VPN support
# Fallback to localhost:1234 if not provided
BASE_URL = os.getenv("LM_STUDIO_BASE_URL", "http://localhost:1234/v1")
client = AsyncOpenAI(base_url=BASE_URL, api_key="lm-studio")

# =============================================================================
# STATE MANAGEMENT (In-Memory)
# =============================================================================

# agents stores: { agent_id: {"role": str, "system_prompt": str} }
agents: Dict[str, Dict[str, str]] = {}

# tasks stores: { task_id: {"status": str, "result": str} }
# status can be: 'running', 'completed', or 'failed'
tasks: Dict[str, Dict[str, Any]] = {}

# =============================================================================
# SUB-AGENT LOGIC (Background Worker)
# =============================================================================

async def _run_agent_task(agent_id: str, task_id: str, prompt: str):
    """
    The background worker that performs the actual LLM inference.
    This runs concurrently without blocking the MCP server.
    """
    agent_profile = agents.get(agent_id)
    if not agent_profile:
        tasks[task_id] = {"status": "failed", "result": "Agent profile not found."}
        return

    try:
        # Format the chat messages using the agent's specific system prompt
        messages = [
            {"role": "system", "content": agent_profile["system_prompt"]},
            {"role": "user", "content": prompt}
        ]

        # Perform the asynchronous API call to LM Studio
        response = await client.chat.completions.create(
            model="local-model",  # LM Studio typically ignores the model name and uses the loaded one
            messages=messages,
            temperature=0.7,
        )

        # Extract result and update state
        result_text = response.choices[0].message.content
        tasks[task_id] = {
            "status": "completed",
            "result": result_text
        }

    except Exception as e:
        # Log the error into the task status for the supervisor to see
        tasks[task_id] = {
            "status": "failed",
            "result": f"Error during execution: {str(e)}"
        }

# =============================================================================
# MCP TOOL DEFINITIONS
# =============================================================================

@mcp.tool()
async def create_agent(role: str, system_prompt: str) -&gt; str:
    """
    Generates a unique sub-agent with a specific role and system prompt.
    Returns the unique agent_id.
    """
    agent_id = f"agent_{uuid.uuid4().hex[:8]}"
    agents[agent_id] = {
        "role": role,
        "system_prompt": system_prompt
    }
    return agent_id

@mcp.tool()
async def assign_task(agent_id: str, prompt: str) -&gt; str:
    """
    Assigns a task to an existing agent. This is non-blocking.
    Returns a task_id that can be used to track progress.
    """
    # Verify agent exists
    if agent_id not in agents:
        return f"Error: Agent {agent_id} does not exist."

    # Generate unique task ID
    task_id = f"task_{uuid.uuid4().hex[:8]}"
    
    # Initialize task status
    tasks[task_id] = {"status": "running", "result": ""}

    # Kick off the background task using asyncio.create_task
    # This allows the tool to return the task_id immediately
    asyncio.create_task(_run_agent_task(agent_id, task_id, prompt))

    return task_id

@mcp.tool()
async def check_task_status(task_id: str) -&gt; str:
    """
    Returns the current status of a task ('running', 'completed', or 'failed').
    """
    task = tasks.get(task_id)
    if not task:
        return "Error: Task ID not found."
    
    return task["status"]

@mcp.tool()
async def retrieve_result(task_id: str) -&gt; str:
    """
    Returns the generated text from the sub-agent if the task is completed.
    """
    task = tasks.get(task_id)
    if not task:
        return "Error: Task ID not found."
    
    if task["status"] == "running":
        return "Task is still running. Please check status later."
    
    if task["status"] == "failed":
        return f"Task failed. Error: {task.get('result')}"

    return task.get("result", "No result found.")

# =============================================================================
# ENTRY POINT
# =============================================================================

if __name__ == "__main__":
    # FastMCP handles the stdio transport and lifecycle automatically
    mcp.run()
```

#### **Scenario A: The "Queue" Behavior (Most Common)**

*   **What happens:** The MCP server will successfully "assign" the task and return a `task_id`.

```python
import os
import uuid
import asyncio
from typing import Dict, Any
from openai import AsyncOpenAI
from mcp.server.fastmcp import FastMCP

# =============================================================================
# INITIALIZATION & CONFIGURATION
# =============================================================================

# Initialize FastMCP server
mcp = FastMCP("Dynamic Supervisor")

# API Configuration
BASE_URL = os.getenv("LM_STUDIO_BASE_URL", "http://localhost:1234/v1")
API_KEY = os.getenv("LM_STUDIO_API_KEY", "lm-studio")
client = AsyncOpenAI(base_url=BASE_URL, api_key=API_KEY)

# CONCURRENCY CONTROL
# This limits how many LLM API calls can happen at the exact same time.
# Default is 3. You can change this via environment variable.
CONCURRENCY_LIMIT = int(os.getenv("MCP_CONCURRENCY_LIMIT", "3"))
semaphore = asyncio.Semaphore(CONCURRENCY_LIMIT)

# =============================================================================
# STATE MANAGEMENT (In-Memory)
# =============================================================================

# agents stores: { agent_id: {"role": str, "system_prompt": str} }
agents: Dict[str, Dict[str, str]] = {}

# tasks stores: { task_id: {"status": str, "result": str} }
tasks: Dict[str, Dict[str, Any]] = {}

# =============================================================================
# SUB-AGENT LOGIC (Background Worker)
# =============================================================================

async def _run_agent_task(agent_id: str, task_id: str, prompt: str):
    """
    The background worker that performs the actual LLM inference.
    Uses a Semaphore to ensure we don't overwhelm the local LLM server.
    """
    agent_profile = agents.get(agent_id)
    if not agent_profile:
        tasks[task_id] = {"status": "failed", "result": "Agent profile not found."}
        return

    try:
        # The Semaphore acts as a gatekeeper. 
        # If the limit is reached, this 'async with' block will pause 
        # execution here until a 'slot' opens up.
        async with semaphore:
            # Format the chat messages using the agent's specific system prompt
            messages = [
                {"role": "system", "content": agent_profile["system_prompt"]},
                {"role": "user", "content": prompt}
            ]

            # Perform the asynchronous API call to LM Studio
            response = await client.chat.completions.create(
                model="local-model", 
                messages=messages,
                temperature=0.7,
            )

            # Extract result
            result_text = response.choices[0].message.content
            tasks[task_id] = {
                "status": "completed",
                "result": result_text
            }

    except Exception as e:
        # If the API call fails or a timeout occurs, log it to the task status
        tasks[task_id] = {
            "status": "failed",
            "result": f"Error during execution: {str(e)}"
        }

# =============================================================================
# MCP TOOL DEFINITIONS
# =============================================================================

@mcp.tool()
async def create_agent(role: str, system_prompt: str) -&gt; str:
    """
    Generates a unique sub-agent with a specific role and system prompt.
    Returns the unique agent_id.
    """
    agent_id = f"agent_{uuid.uuid4().hex[:8]}"
    agents[agent_id] = {
        "role": role,
        "system_prompt": system_prompt
    }
    return agent_id

@mcp.tool()
async def assign_task(agent_id: str, prompt: str) -&gt; str:
    """
    Assigns a task to an existing agent. This is non-blocking.
    Returns a task_id that can be used to track progress.
    """
    if agent_id not in agents:
        return f"Error: Agent {agent_id} does not exist."

    task_id = f"task_{uuid.uuid4().hex[:8]}"
    
    # Initialize task status as 'running' immediately.
    # If it hits the semaphore, it will stay 'running' while waiting in line.
    tasks[task_id] = {"status": "running", "result": ""}

    # Kick off the background task
    asyncio.create_task(_run_agent_task(agent_id, task_id, prompt))

    return task_id

@mcp.tool()
async def check_task_status(task_id: str) -&gt; str:
    """
    Returns the current status of a task ('running', 'completed', or 'failed').
    """
    task = tasks.get(task_id)
    if not task:
        return "Error: Task ID not found."
    
    return task["status"]

@mcp.tool()
async def retrieve_result(task_id: str) -&gt; str:
    """
    Returns the generated text from the sub-agent if the task is completed.
    """
    task = tasks.get(task_id)
    if not task:
        return "Error: Task ID not found."
    
    if task["status"] == "running":
        return "Task is still running (or waiting in the concurrency queue)."
    
    if task["status"] == "failed":
        return f"Task failed. Error: {task.get('result')}"

    return task.get("result", "No result found.")

# =============================================================================
# ENTRY POINT
# =============================================================================

if __name__ == "__main__":
    mcp.run()
```
